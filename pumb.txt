#include <Stepper.h>

// تعريف دبابيس المحرك
int IN1 = 34;
int IN2 = 35;
int IN3 = 36;
int IN4 = 37;

// عدد الخطوات لكل دورة كاملة (حسب المحرك)
const int steps_per_revolution = 200;  // يمكن تعديلها حسب المحرك

// إنشاء كائن من المكتبة للتحكم في المحرك
Stepper motor(steps_per_revolution, IN1, IN2, IN3, IN4);

// المتغير للتحكم في السرعة
int motor_speed = 370;  // تعيين السرعة إلى الحد الأقصى (الدورات في الدقيقة)

void setup() {
  // تعيين السرعة للمحرك
  motor.setSpeed(motor_speed);  // تعيين السرعة (بـ RPM)
}

void loop() {
  // تدوير المحرك دورة كاملة باتجاه عقارب الساعة فقط
  motor.step(-1);  // تحريك المحرك خطوة واحدة باتجاه عقارب الساعة

  //delay(100);  // تأخير قصير بين كل دورة (أقل من 100 مليل ثانية لتحسين السرعة)
}
